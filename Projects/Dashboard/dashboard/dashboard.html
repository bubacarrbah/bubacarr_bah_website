import dash

import plotly.express as px
import pandas as pd
from sklearn.cluster import KMeans


df = pd.read_csv('typing_competition.csv')  # Replace with your CSV file path

# Data Cleaning
df['ACCURACY'] = df['ACCURACY'].str.rstrip('%').astype('float') / 100.0
df['SPEED'] = df['SPEED'].str.rstrip('wpm').astype('float')
df['TOTAL SCORE'] = df['TOTAL SCORE'].str.rstrip(' points').astype('float')
df['GR.NO'] = df['GR.NO'].replace('-', pd.NA)


df['SPEED_ACCURACY'] = df['SPEED'] * df['ACCURACY']  

# Cluster students
kmeans = KMeans(n_clusters=3, random_state=42)
df['PERFORMANCE_CLUSTER'] = kmeans.fit_predict(df[['SPEED', 'ACCURACY']])
df['PERFORMANCE_CATEGORY'] = df['PERFORMANCE_CLUSTER'].map({0: 'Low', 1: 'Medium', 2: 'High'})

# Predictive Insights
@app.callback(
    [Output('predicted-score', 'children'),
     Output('prediction-explanation', 'children'),
     Output('recommendations', 'children'),
     Output('prediction-gauge', 'figure')],
    [Input('speed-input', 'value'),
     Input('accuracy-input', 'value')]
)
def update_predictions(speed, accuracy):
    # Calculate the predicted score using 60% speed and 40% accuracy
    predicted_score = (0.6 * speed) + (0.4 * accuracy)
    
    # Explanation of factors
    explanation = (
        f"Your predicted score is calculated as 60% of speed ({0.6 * speed:.2f}) "
        f"and 40% of accuracy ({0.4 * accuracy:.2f})."
    )
    
    # Personalized recommendations
    if speed < 30 and accuracy < 70:
        recommendations = "Focus on both speed and accuracy drills to improve your score."
    elif speed < 30:
        recommendations = "Focus on speed drills to type faster."
    elif accuracy < 70:
        recommendations = "Focus on accuracy drills to reduce errors."
    else:
        recommendations = "Keep up the good work! You're doing great."
    
    # Score Predictor
    gauge_fig = {
        'data': [
            {
                'type': 'indicator',
                'mode': 'gauge+number',
                'value': predicted_score,
                'title': {'text': 'Predicted Total Score'},
                'gauge': {
                    'axis': {'range': [0, 100]},
                    'steps': [
                        {'range': [0, 50], 'color': 'lightgray'},
                        {'range': [50, 80], 'color': 'gray'},
                        {'range': [80, 100], 'color': 'darkgray'}
                    ],
                    'threshold': {
                        'line': {'color': 'red', 'width': 4},
                        'thickness': 0.75,
                        'value': predicted_score
                    }
                }
            }
        ],
        'layout': {'height': 300}
    }
    
    return (
        f"Predicted Total Score: {predicted_score:.2f}",
        explanation,
        recommendations,
        gauge_fig
    )


if __name__ == '__main__':
    app.run_server(debug=True)
